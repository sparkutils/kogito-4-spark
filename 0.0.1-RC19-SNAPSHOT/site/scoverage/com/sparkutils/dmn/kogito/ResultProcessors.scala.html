178   def setStatuses(tres: GenericInternalRow, res: org.kie.dmn.api.core.DMNResult): GenericInternalRow =
179     evalStatus.foldLeft(tres) {
180       case (row, (field, i)) =>
181         val decisionName = field.name.dropRight(evalStatusEnding.length)
182         val dr = res.getDecisionResultByName(decisionName)
183         if (dr ne null) {
184           row.update(i, dr.getEvaluationStatus match {
185             case DecisionEvaluationStatus.NOT_EVALUATED => NOT_EVALUATED
186             case DecisionEvaluationStatus.EVALUATING => EVALUATING
187             case DecisionEvaluationStatus.SUCCEEDED => SUCCEEDED
188             case DecisionEvaluationStatus.SKIPPED if dr.hasErrors => SKIPPED_ERROR
189             case DecisionEvaluationStatus.SKIPPED => SKIPPED_WARN
190             case DecisionEvaluationStatus.FAILED => FAILED
191           })
192         } else {
193           row.update(i, NOT_FOUND)
194         }
195         row
