103 case class KogitoDMNContext(ctx: org.kie.dmn.api.core.DMNContext) extends DMNContext {
104 
105   def set(path: DMNContextPath, data: Any): Unit = {
106     val bits = path.asInstanceOf[KogitoDMNContextPath].path.split('.')
107     val starter =
108       ctx.get(bits(0)) match {
109         case _ if bits.length == 1 =>
110           ctx.set(bits.head, data)
111           return
112         case null if bits.length > 1 =>
113           val n = new util.HashMap[String, Object]()
114           ctx.set(bits.head, n)
115           n
116         case t: MAP =>
117           t
118         case _ =>
119           ctx.set(bits.head, data)
120           return
121       }
122 
123     val remaining =
124       if (data.isInstanceOf[MAP])
125         bits.drop(1)
126       else
127         bits.drop(1).dropRight(1)
128 
129     remaining.foldLeft(starter){
130       (map, pathBit) =>
131         val n =
132           map.get(pathBit) match {
133             case null => new util.HashMap[String, Object]()
134             case t: MAP => t
135           }
136         map.put(pathBit, n)
137         n
138     }
139 
140     def updateContext(bits: Seq[String], map: MAP): MAP =
141       if (bits.size == 1) {
142         map.put(bits.head, data.asInstanceOf[Object])
143         map
144       } else
145         updateContext(bits.drop(1), map.get(bits.head).asInstanceOf[MAP])
146 
147     updateContext(bits.drop(1).toVector, starter)
148   }
149 }
