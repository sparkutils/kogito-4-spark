64   override def providerForType(inputField: DMNInputField, debug: Boolean, dmnConfiguration: DMNConfiguration): DMNContextProvider[_] = {
65     val (path, expr) = (KogitoDMNContextPath(inputField.contextPath), inputField.defaultExpr)

78           case DoubleType => SimpleContextProvider[Double](path, inputField.stillSetWhenNull, expr)
79           case FloatType => SimpleContextProvider[Float](path, inputField.stillSetWhenNull, expr)
80           case BinaryType => SimpleContextProvider[Array[Byte]](path, inputField.stillSetWhenNull, expr)
81           case ByteType => SimpleContextProvider[Byte](path, inputField.stillSetWhenNull, expr)
82           case ShortType => SimpleContextProvider[Short](path, inputField.stillSetWhenNull, expr)
83           case DateType => SimpleContextProvider[LocalDate](path, inputField.stillSetWhenNull, expr,
84             Some(((t: Any) => DateTimeUtils.daysToLocalDate(t.asInstanceOf[Int]),
85               (codegen, input) => s"org.apache.spark.sql.catalyst.util.DateTimeUtils.daysToLocalDate((int)$input)"))
86           ) // an int
87           case TimestampType => SimpleContextProvider[LocalDateTime](path, inputField.stillSetWhenNull, expr,

89               (codegen, input) => s"org.apache.spark.sql.catalyst.util.DateTimeUtils.microsToLocalDateTime((long)$input)"))
90           ) // a long
91           case _: DecimalType => SimpleContextProvider[java.math.BigDecimal](path, inputField.stillSetWhenNull, expr,
92             Some(((t: Any) => t.asInstanceOf[Decimal].toJavaBigDecimal,
93               (codegen, input) => s"((${classOf[Decimal].getName})$input).toJavaBigDecimal()"))
94           )
95           case structType: StructType => ContextInterfaces.structProvider(structType, path, expr, inputField.stillSetWhenNull, config)
96           case mapType: MapType => ContextInterfaces.mapProvider(mapType, path, expr, inputField.stillSetWhenNull, config)
97           case arrayType: ArrayType => ContextInterfaces.arrayProvider(arrayType, path, expr, inputField.stillSetWhenNull, config)
98           // calendar interval?
99           case t => throw new DMNException(s"Provider type $t is not supported")
100         }
101       case _ =>
102         utils.loadUnaryContextProvider(inputField.providerType, path, expr)
103     }

127 case class KogitoDMNContext(ctx: org.kie.dmn.api.core.DMNContext) extends DMNContext {
128 
129   def set(path: DMNContextPath, data: Any): Unit = {
130     val bits = path.asInstanceOf[KogitoDMNContextPath].path.split('.')
131     val starter =
132       ctx.get(bits(0)) match {
133         case _ if bits.length == 1 =>
134           ctx.set(bits.head, data)
135           return
136         case null if bits.length > 1 =>
137           val n = new util.HashMap[String, Object]()
138           ctx.set(bits.head, n)
139           n
140         case t: MAP =>
141           t
142         case _ => // TODO log warn
143           ctx.set(bits.head, data)
144           return
145       }
146 
147     // top is the root context, bottom is the place we'd store things
148     bits.drop(1).dropRight(1).foldLeft(starter){
149       (map, pathBit) =>
150         map match {
151           case t: MAP =>
152             val n =
153               t.get(pathBit) match {
154                 case null => new util.HashMap[String, Object]()
155                 case t: MAP => t
156                 case _ => new util.HashMap[String, Object]()
157               }
158             t.put(pathBit, n)
159             n
160           case _ => map
161         }
162     }
163 
164     def updateContext(bits: Seq[String], map: MAP): MAP =
165       if (bits.size == 1) {
166         map.put(bits.head, data.asInstanceOf[Object])
167         map
168       } else
169         updateContext(bits.drop(1), map.get(bits.head).asInstanceOf[MAP])
170 
171     ctx.set(bits.head, updateContext(bits.drop(1).toVector, starter)) // drop the 1st as that's for the root context
172   }
173 }
