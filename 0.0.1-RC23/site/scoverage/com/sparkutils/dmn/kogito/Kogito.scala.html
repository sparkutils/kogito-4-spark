62   override def providerForType(inputField: DMNInputField, debug: Boolean, dmnConfiguration: DMNConfiguration): DMNContextProvider[_] = {

103 case class KogitoDMNContext(ctx: org.kie.dmn.api.core.DMNContext) extends DMNContext {
104 
105   def set(path: DMNContextPath, data: Any): Unit = {
106     val bits = path.asInstanceOf[KogitoDMNContextPath].path.split('.')
107     val starter =
108       ctx.get(bits(0)) match {
109         case _ if bits.length == 1 =>
110           // top level direct entries only (map or otherwise)
111           ctx.set(bits.head, data)
112           return
113         case t: MAP =>
114           t
115         case _ =>
116           // any other top level field must be overwritten
117           val n = new util.HashMap[String, Object]()
118           ctx.set(bits.head, n)
119           n
120       }
121 
122     val remaining =
123       if (data.isInstanceOf[MAP])
124         bits.drop(1)
125       else
126         bits.drop(1).dropRight(1)
127 
128     remaining.foldLeft(starter){
129       (map, pathBit) =>
130         val n =
131           map.get(pathBit) match {
132             case null => new util.HashMap[String, Object]()
133             case t: MAP => t
134           }
135         map.put(pathBit, n)
136         n
137     }
138 
139     def updateContext(bits: Seq[String], map: MAP): MAP =
140       if (bits.size == 1) {
141         map.put(bits.head, data.asInstanceOf[Object])
142         map
143       } else
144         updateContext(bits.drop(1), map.get(bits.head).asInstanceOf[MAP])
145 
146     updateContext(bits.drop(1).toVector, starter)
147   }
148 }
