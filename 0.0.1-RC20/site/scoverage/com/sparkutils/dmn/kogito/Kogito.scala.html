60   override def providerForType(inputField: DMNInputField, debug: Boolean, dmnConfiguration: DMNConfiguration): DMNContextProvider[_] = {
61     val (path, expr) = (KogitoDMNContextPath(inputField.contextPath), inputField.defaultExpr)

63     val config = configMap(dmnConfiguration)
64 
65     inputField.providerType match {
66       case "" => ContextProviderProxy(path, inputField.stillSetWhenNull, expr, config)

77   override def resultProviderForType(resultProviderType: String, debug: Boolean, dmnConfiguration: DMNConfiguration): DMNResultProvider =
78     resultProviderType match {

97 case class KogitoDMNContext(ctx: org.kie.dmn.api.core.DMNContext) extends DMNContext {
98 
99   def set(path: DMNContextPath, data: Any): Unit = {
100     val bits = path.asInstanceOf[KogitoDMNContextPath].path.split('.')
101     val starter =
102       ctx.get(bits(0)) match {
103         case _ if bits.length == 1 =>
104           ctx.set(bits.head, data)
105           return
106         case null if bits.length > 1 =>
107           val n = new util.HashMap[String, Object]()
108           ctx.set(bits.head, n)
109           n
110         case t: MAP =>
111           t
112         case _ => // TODO log warn
113           ctx.set(bits.head, data)
114           return
115       }
116 
117     // top is the root context, bottom is the place we'd store things
118     bits.drop(1).dropRight(1).foldLeft(starter){
119       (map, pathBit) =>
120         map match {
121           case t: MAP =>
122             val n =
123               t.get(pathBit) match {
124                 case null => new util.HashMap[String, Object]()
125                 case t: MAP => t
126                 case _ => new util.HashMap[String, Object]()
127               }
128             t.put(pathBit, n)
129             n
130           case _ => map
131         }
132     }
133 
134     def updateContext(bits: Seq[String], map: MAP): MAP =
135       if (bits.size == 1) {
136         map.put(bits.head, data.asInstanceOf[Object])
137         map
138       } else
139         updateContext(bits.drop(1), map.get(bits.head).asInstanceOf[MAP])
140 
141     ctx.set(bits.head, updateContext(bits.drop(1).toVector, starter)) // drop the 1st as that's for the root context
142   }
143 }
