63   override def providerForType(inputField: DMNInputField, debug: Boolean, dmnConfiguration: DMNConfiguration): DMNContextProvider[_] = {
64     val (path, expr) = (KogitoDMNContextPath(inputField.contextPath), inputField.defaultExpr)
65 
66     val config = configMap(dmnConfiguration)
67 
68     inputField.providerType match {
69       case "JSON" => KogitoJSONContextProvider(path, inputField.stillSetWhenNull, expr)
70       case t if Try(DataType.fromDDL(t)).isSuccess =>
71         val dataType = DataType.fromDDL(t)
72         dataType match {
73           case StringType => StringContextProvider(path, inputField.stillSetWhenNull, expr)
74           case IntegerType => SimpleContextProvider[Integer](path, inputField.stillSetWhenNull, expr)
75           case LongType => SimpleContextProvider[Long](path, inputField.stillSetWhenNull, expr)
76           case BooleanType => SimpleContextProvider[Boolean](path, inputField.stillSetWhenNull, expr)
77           case DoubleType => SimpleContextProvider[Double](path, inputField.stillSetWhenNull, expr)
78           case FloatType => SimpleContextProvider[Float](path, inputField.stillSetWhenNull, expr)
79           case BinaryType => SimpleContextProvider[Array[Byte]](path, inputField.stillSetWhenNull, expr)
80           case ByteType => SimpleContextProvider[Byte](path, inputField.stillSetWhenNull, expr)
81           case ShortType => SimpleContextProvider[Short](path, inputField.stillSetWhenNull, expr)
82           case DateType => SimpleContextProvider[LocalDate](path, inputField.stillSetWhenNull, expr,
83             Some(((t: Any) => DateTimeUtils.daysToLocalDate(t.asInstanceOf[Int]),
84               (codegen, input) => s"org.apache.spark.sql.catalyst.util.DateTimeUtils.daysToLocalDate((int)$input)"))
85           ) // an int
86           case TimestampType => SimpleContextProvider[LocalDateTime](path, inputField.stillSetWhenNull, expr,
87             Some(((t: Any) => DateTimeUtils.microsToLocalDateTime(t.asInstanceOf[Long]),
88               (codegen, input) => s"org.apache.spark.sql.catalyst.util.DateTimeUtils.microsToLocalDateTime((long)$input)"))
89           ) // a long
90           case _: DecimalType => SimpleContextProvider[java.math.BigDecimal](path, inputField.stillSetWhenNull, expr,
91             Some(((t: Any) => t.asInstanceOf[Decimal].toJavaBigDecimal,
92               (codegen, input) => s"((${classOf[Decimal].getName})$input).toJavaBigDecimal()"))
93           )
94           case structType: StructType => ContextInterfaces.structProvider(structType, path, expr, inputField.stillSetWhenNull, config)
95           case mapType: MapType => ContextInterfaces.mapProvider(mapType, path, expr, inputField.stillSetWhenNull, config)
96           case arrayType: ArrayType => ContextInterfaces.arrayProvider(arrayType, path, expr, inputField.stillSetWhenNull, config)
97           // calendar interval?
98           case t => throw new DMNException(s"Provider type $t is not supported")
99         }
100       case _ =>
101         utils.loadUnaryContextProvider(inputField.providerType, path, expr)
102     }
103 
104   }

126 case class KogitoDMNContext(ctx: org.kie.dmn.api.core.DMNContext) extends DMNContext {
127 
128   def set(path: DMNContextPath, data: Any): Unit = {
129     val bits = path.asInstanceOf[KogitoDMNContextPath].path.split('.')
130     val starter =
131       ctx.get(bits(0)) match {
132         case _ if bits.length == 1 =>
133           ctx.set(bits.head, data)
134           return
135         case null if bits.length > 1 =>
136           val n = new util.HashMap[String, Object]()
137           ctx.set(bits.head, n)
138           n
139         case t: MAP =>
140           t
141         case _ => // TODO log warn
142           ctx.set(bits.head, data)
143           return
144       }
145 
146     // top is the root context, bottom is the place we'd store things
147     bits.drop(1).dropRight(1).foldLeft(starter){
148       (map, pathBit) =>
149         map match {
150           case t: MAP =>
151             val n =
152               t.get(pathBit) match {
153                 case null => new util.HashMap[String, Object]()
154                 case t: MAP => t
155                 case _ => new util.HashMap[String, Object]()
156               }
157             t.put(pathBit, n)
158             n
159           case _ => map
160         }
161     }
162 
163     def updateContext(bits: Seq[String], map: MAP): MAP =
164       if (bits.size == 1) {
165         map.put(bits.head, data.asInstanceOf[Object])
166         map
167       } else
168         updateContext(bits.drop(1), map.get(bits.head).asInstanceOf[MAP])
169 
170     ctx.set(bits.head, updateContext(bits.drop(1).toVector, starter)) // drop the 1st as that's for the root context
171   }
172 }
